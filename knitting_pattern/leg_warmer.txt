stitch rib1x1 = K 1, P 1;
pattern cuff(width){
    repeat 5 {
        rib1x1 * 10, K 1;
    }
}

pattern leg(width){
   repeat 10 {
        K 1 fill;
    }
}

pattern legwarmer(width, size) {
    print("using circular needles");
    cast_on width;
    work(cuff(width));
    print("switch to main yarn");
    K 1 fill;  
    work(leg(width));
    print("switch to cuff yarn");
    work(cuff(width));
    bind_off width;
}

execute(legwarmer(21 , 0));
